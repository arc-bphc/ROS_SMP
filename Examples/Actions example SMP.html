Actions¶

ROS actions are the best way to implement interfaces to time-extended, goal-oriented behaviors. While services are synchronous, actions are asynchro nous. Similar to the request and response of a service, an action uses a goal to initiate a behavior and sends a result when the behavior is complete. But the action further uses feedback to provide updates on the behaviors progress toward the goal and also allows for goals to be canceled. Actions are themselves implemented using topics. An action is essentially a higher-level protocol that specifies how a set of topics (goal, result, feedback, etc.) should be used in combination.

In order to find which actions are available on a robot, you must do a rostopic list.

Calling an action server¶

The message of an action server is divided into three parts: the goal, the result, and the feedback.

Due to the fact that calling an action server does not interrupt your thread, action servers provide a message called the feedback. The feedback is a message that the action server generates every once in a while to indicate how the action is going (informing the caller of the status of the requested action). It is generated while the action is in progress.

Defined a class to encapsulate everything¶

  • Constructor creates the action server (name, action message file, goal callback, Flase (Autostart) and starts it
  • define a goal callback (this is the main part, working towards a result)
    • publish feedbacks at regular intervals when required
    • publish the result after the computation is done
  • Finally, initalise a node, an object of the class in main and do rospy.spin()

Example question:¶

Write an action server that receives a threshold distance from an obstacle (directly in front of the bot) as a goal and stops the turtlebot as soon as it is crossed.

Feedback is the current distance from the obstacle in front.

Result is a string that is sent back after the bot is stopped.

#! /usr/bin/env python
import rospy
import actionlib
from action_tut.msg import CrashAction, CrashResult
from geometry_msgs.msg import Twist
from sensor_msgs.msg import LaserScan

class SimpleActionServer:
    def __init__(self):
        self.result = CrashResult()
        self.current_distance = 1000
        self.velocity = Twist()
        self.laser_sub = rospy.Subscriber('/scan', LaserScan, self.laser_callback, queue_size= 1 )
        self.vel_pub  = rospy.Publisher('/cmd_vel', Twist,latch= True, queue_size= 1)
        self._as = actionlib.SimpleActionServer('crash_service', CrashAction, self.goal_callback, False)
        self._as.start()
        rospy.loginfo("Action server: Action server started! ")

    def laser_callback(self, msg):
        # set the current distance as the front laser value
        self.current_distance = msg.ranges[0]

    def goal_callback(self,goal):
        # send the result once the treshold is crossed and the bot is stopped
        print(goal.threshold)
        while self.current_distance > goal.threshold:
            rospy.loginfo("Current ditance: %f" %self.current_distance)
            rospy.sleep(0.5)
        
        #crossed the threshold
        rospy.loginfo(" Crossed threshold at: %f" %self.current_distance)
        self.velocity.linear.x = 0
        self.vel_pub.publish(self.velocity)
        rospy.loginfo("Action server: Stopping the robot")
        self.result.result = " Stopped the robot!"
        self._as.set_succeeded(self.result)
        

if __name__ == '__main__':
    rospy.init_node("crash_server_node")
    SimpleActionServer()
    rospy.spin()

import rospy
import actionlib
from action_tut.msg import CrashAction, CrashResult, CrashFeedback
from geometry_msgs.msg import Twist
from sensor_msgs.msg import LaserScan

class SimpleActionServer:
    def __init__(self):
        self.result = CrashResult()
        self.feedback = CrashFeedback()
        self.current_distance = 1000
        self.velocity = Twist()
        self.laser_sub = rospy.Subscriber('/scan', LaserScan, self.laser_callback, queue_size= 1 )
        self.vel_pub  = rospy.Publisher('/cmd_vel', Twist,latch= True, queue_size= 1)
        self._as = actionlib.SimpleActionServer('crash_service', CrashAction, self.goal_callback, False)
        self._as.start()
        rospy.loginfo("Action server: Action server started! ")

    def laser_callback(self, msg):
        # set the current distance as the front laser value
        self.current_distance = msg.ranges[0]

    def goal_callback(self,goal):
        # send the result once the treshold is crossed and the bot is stopped
        print(goal.threshold)
        while self.current_distance > goal.threshold:
            rospy.loginfo("Current ditance: %f" %self.current_distance)
            self.feedback.current_distance = self.current_distance
            self._as.publish_feedback(self.feedback)
            rospy.sleep(0.5)
        
        #crossed the threshold
        rospy.loginfo(" Crossed threshold at: %f" %self.current_distance)
        self.velocity.linear.x = 0
        self.vel_pub.publish(self.velocity)
        rospy.loginfo("Action server: Stopping the robot")
        self.result.result = " Stopped the robot!"

    rospy.init_node("crash_server_node")
    SimpleActionServer()
    rospy.spin()

#! /usr/bin/env python
import rospy
import actionlib
from geometry_msgs.msg import Twist
from action_tut.msg import CrashAction, CrashGoal


"""
class SimpleGoalState:
    PENDING = 0
    ACTIVE = 1
    DONE = 2
    WARN = 3
    ERROR = 4

"""
# We create some constants with the corresponing vaules from the SimpleGoalState class
PENDING = 0
ACTIVE = 1
DONE = 2
WARN = 3
ERROR = 4

def feedback_callback(feedback):
    print(feedback.current_distance)


rospy.init_node('simple_action_client_node')
goal = CrashGoal()
twist = Twist()
goal.threshold = 0.6

# set the turtlebot into motion
vel_pub = rospy.Publisher('/cmd_vel', Twist, latch= True, queue_size=1)

# make the client
client = actionlib.SimpleActionClient('/crash_service', CrashAction)
client.wait_for_server()


# move forward
rospy.loginfo("Moving the bot forward")
twist.linear.x = 0.4
vel_pub.publish(twist)

#send the goal
client.send_goal(goal, feedback_cb=feedback_callback)
# can do other stuff here

state_result = client.get_state()
rate = rospy.Rate(1)

rospy.loginfo("state_result: "+str(state_result))

while state_result < DONE:
    rospy.loginfo("Doing Stuff while waiting for the Server to give a result....")
    rate.sleep()
    state_result = client.get_state()
    
    rospy.loginfo("state_result: "+str(state_result))
    
rospy.loginfo("[Result] State: "+str(state_result))
if state_result == ERROR:
    rospy.logerr("Something went wrong in the Server Side")
if state_result == WARN:
    rospy.logwarn("There is a warning in the Server Side")
